{{- /* Params: Resource: Hugo image resource object (Optional, for processing) Widths: Slice of widths to generate for srcset Attributes: Map of HTML attributes for the tag Enhanced: Outputs WebP inside a element when possible. */ -}} {{- $resource := .Resource -}} {{- $widths := .Widths -}} {{- $attributes := .Attributes | default dict -}} {{- /* Check if we can process this image (not SVG, not remote without processing) */ -}} {{- $processable := and $widths $resource (reflect.IsImageResourceProcessable $resource) -}} {{- if $processable -}} {{- /* Generate WebP srcset */ -}} {{- $webpSrcset := slice -}} {{- $fallbackSrcset := slice -}} {{- range $widths -}} {{- if lt . $resource.Width -}} {{- $resized := $resource.Resize (printf "%dx" .) -}} {{- $webpResized := $resource.Resize (printf "%dx webp" .) -}} {{- $webpSrcset = $webpSrcset | append (printf "%s %dw" $webpResized.RelPermalink .) -}} {{- $fallbackSrcset = $fallbackSrcset | append (printf "%s %dw" $resized.RelPermalink .) -}} {{- end -}} {{- end -}} {{- /* Original size */ -}} {{- $webpFull := $resource.Resize (printf "%dx webp" $resource.Width) -}} {{- $webpSrcset = $webpSrcset | append (printf "%s %dw" $webpFull.RelPermalink $resource.Width) -}} {{- $fallbackSrcset = $fallbackSrcset | append (printf "%s %dw" $resource.RelPermalink $resource.Width) -}} {{- $alt := index $attributes "alt" -}} {{- $title := index $attributes "title" -}} {{- $class := index $attributes "class" -}} {{- $loading := index $attributes "loading" -}} {{- $sizes := index $attributes "sizes" -}} {{- $width := index $attributes "width" -}} {{- $height := index $attributes "height" -}} {{- else -}} {{- /* Fallback: standard tag when processing isn't possible (SVG, remote, etc.) */ -}} {{- end -}}
